def preprocess_text(text):
    """Clean and preprocess text data"""
    text = text.lower()
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text

def build_rnn_model(vocab_size, embedding_dim=100, max_length=200, num_classes=5):
    """Build RNN model using LSTM layers"""
    model = Sequential([
        Embedding(vocab_size, embedding_dim, input_length=max_length),
        LSTM(128, dropout=0.2, recurrent_dropout=0.2, return_sequences=True),
        LSTM(64, dropout=0.2, recurrent_dropout=0.2),
        Dense(32, activation='relu'),
        Dropout(0.3),
        Dense(num_classes, activation='softmax')
    ])

    model.compile(
        optimizer='adam',
        loss='categorical_crossentropy',
        metrics=['accuracy']
    )
    return model

def build_cnn_model(vocab_size, embedding_dim=100, max_length=200, num_classes=5):
    """Build CNN model for text classification"""
    model = Sequential([
        Embedding(vocab_size, embedding_dim, input_length=max_length),
        Conv1D(128, 5, activation='relu'),
        MaxPooling1D(5),
        Conv1D(64, 5, activation='relu'),
        MaxPooling1D(5),
        Conv1D(32, 5, activation='relu'),
        GlobalMaxPooling1D(),
        Dense(64, activation='relu'),
        Dropout(0.3),
        Dense(num_classes, activation='softmax')
    ])

    model.compile(
        optimizer='adam',
        loss='categorical_crossentropy',
        metrics=['accuracy']

class NGramModel:
    """N-gram language model implementation"""

    def __init__(self, n):
        self.n = n
        self.ngrams = defaultdict(Counter)
        self.vocab = set()

    def train(self, words):
        """Train the n-gram model"""
        self.vocab = set(words)

        # Add start and end tokens
        padded_words = ['<START>'] * (self.n - 1) + words + ['<END>']

        for i in range(len(padded_words) - self.n + 1):
            context = tuple(padded_words[i:i + self.n - 1])
            next_word = padded_words[i + self.n - 1]
            self.ngrams[context][next_word] += 1

    def get_probability(self, context, word):
        """Calculate probability of word given context"""
        context = tuple(context)
        if context not in self.ngrams:
            return 1e-10  # Smoothing for unseen contexts

        total_count = sum(self.ngrams[context].values())
        word_count = self.ngrams[context][word]

        # Add-one smoothing
        return (word_count + 1) / (total_count + len(self.vocab))

    def calculate_perplexity(self, test_words):
        """Calculate perplexity on test data"""
        padded_words = ['<START>'] * (self.n - 1) + test_words + ['<END>']
        log_prob_sum = 0
        word_count = 0

        for i in range(len(padded_words) - self.n + 1):
            context = padded_words[i:i + self.n - 1]
            next_word = padded_words[i + self.n - 1]

            prob = self.get_probability(context, next_word)
            log_prob_sum += math.log(prob)
            word_count += 1

        if word_count == 0:
            return float('inf')

        avg_log_prob = log_prob_sum / word_count
        perplexity = math.exp(-avg_log_prob)
        return perplexity

def calculate_rnn_perplexity(model, X_test, y_test):
    """Calculate perplexity for RNN language model"""
    predictions = model.predict(X_test, verbose=0)
    log_prob_sum = 0

    for i in range(len(y_test)):
        true_word_idx = y_test[i]
        predicted_probs = predictions[i]

        if true_word_idx < len(predicted_probs):
            prob = predicted_probs[true_word_idx]
            prob = max(prob, 1e-10)  # Avoid log(0)
            log_prob_sum += math.log(prob)

    avg_log_prob = log_prob_sum / len(y_test)
    perplexity = math.exp(-avg_log_prob)
    return perplexity

def generate_text(model, seed_text, num_words=20):
    """Generate text using trained RNN language model"""
    seed_words = seed_text.lower().split()
    generated = seed_words.copy()

    for _ in range(num_words):
        # Prepare input sequence
        if len(generated) >= sequence_length:
            input_seq = generated[-sequence_length:]
        else:
            input_seq = ['<UNK>'] * (sequence_length - len(generated)) + generated

        # Convert to indices
        input_indices = [word_to_index.get(word, 0) for word in input_seq]
        input_array = np.array([input_indices])

        # Predict next word
        predictions = model.predict(input_array, verbose=0)[0]

        # Sample from top predictions
        top_indices = np.argsort(predictions)[-10:]
        probs = predictions[top_indices]
        probs = probs / np.sum(probs)

        next_word_idx = np.random.choice(top_indices, p=probs)
        next_word = index_to_word.get(next_word_idx, '<UNK>')

        if next_word != '<UNK>':
            generated.append(next_word)

    return ' '.join(generated)
