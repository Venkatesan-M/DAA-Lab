class FeatureExtractor:
    """Extract MFCC features from audio signals"""

    def __init__(self, sample_rate=16000, n_mfcc=13, n_fft=512, hop_length=160):
        self.sample_rate = sample_rate
        self.n_mfcc = n_mfcc
        self.n_fft = n_fft
        self.hop_length = hop_length

    def extract_mfcc(self, audio_path):
        """Extract MFCC features from audio file"""
        try:
            audio, sr = librosa.load(audio_path, sr=self.sample_rate)

            # Pre-emphasis filter
            pre_emphasis = 0.97
            audio = np.append(audio[0], audio[1:] - pre_emphasis * audio[:-1])

            # Extract MFCC features
            mfcc = librosa.feature.mfcc(y=audio, sr=sr, n_mfcc=self.n_mfcc,
                                         n_fft=self.n_fft, hop_length=self.hop_length)

            # Compute delta and delta-delta coefficients
            mfcc_delta = librosa.feature.delta(mfcc)
            mfcc_delta2 = librosa.feature.delta(mfcc, order=2)

            # Concatenate features
            features = np.vstack([mfcc, mfcc_delta, mfcc_delta2])

            return features.T
        except Exception as e:
            print(f"Error processing {audio_path}: {str(e)}")
            return None

    def extract_statistical_features(self, mfcc_features):
        """Compute statistical measures from MFCC sequence"""
        mean = np.mean(mfcc_features, axis=0)
        std = np.std(mfcc_features, axis=0)
        max_val = np.max(mfcc_features, axis=0)
        min_val = np.min(mfcc_features, axis=0)

        return np.concatenate([mean, std, max_val, min_val])

class DatasetPreparator:
    """Prepare Google Speech Commands dataset for training"""

    def __init__(self, base_path, commands, max_samples_per_command=500):
        self.base_path = base_path
        self.commands = commands
        self.max_samples = max_samples_per_command
        self.feature_extractor = FeatureExtractor()

    def load_data(self):
        """Load audio files and extract features"""
        X = []
        y = []

        for idx, command in enumerate(self.commands):
            command_path = os.path.join(self.base_path, command)

            if not os.path.exists(command_path):
                print(f"Warning: Directory for '{command}' not found")
                continue

            audio_files = [f for f in os.listdir(command_path) if f.endswith('.wav')]
            audio_files = audio_files[:self.max_samples]

            print(f"Processing '{command}': {len(audio_files)} samples")

            for audio_file in audio_files:
                audio_path = os.path.join(command_path, audio_file)
                mfcc = self.feature_extractor.extract_mfcc(audio_path)

                if mfcc is not None:
                    stat_features = self.feature_extractor.extract_statistical_features(mfcc)
                    X.append(stat_features)
                    y.append(idx)

        return np.array(X), np.array(y)

    def split_data(self, X, y, train_ratio=0.8):
        """Split data into training and testing sets"""
        indices = np.random.permutation(len(X))
        train_size = int(len(X) * train_ratio)

        train_indices = indices[:train_size]
        test_indices = indices[train_size:]

        return (X[train_indices], y[train_indices],
                X[test_indices], y[test_indices])

class IsolatedWordRecognizer:
    """DNN-based speech recognition system"""

    def __init__(self):
        self.model = None
        self.scaler = StandardScaler()
        self.commands = []
        self.history = None

    def build_model(self, input_dim, num_classes):
        """Build deep neural network architecture"""
        from tensorflow.keras import layers, models, regularizers

        model = models.Sequential([
            layers.Dense(256, activation='relu', input_shape=(input_dim,),
                        kernel_regularizer=regularizers.l2(0.001)),
            layers.BatchNormalization(),
            layers.Dropout(0.3),

            layers.Dense(128, activation='relu',
                        kernel_regularizer=regularizers.l2(0.001)),
            layers.BatchNormalization(),
            layers.Dropout(0.3),

            layers.Dense(64, activation='relu',
                        kernel_regularizer=regularizers.l2(0.001)),
            layers.BatchNormalization(),
            layers.Dropout(0.2),

            layers.Dense(num_classes, activation='softmax')
        ])

        model.compile(
            optimizer='adam',
            loss='sparse_categorical_crossentropy',
            metrics=['accuracy']
        )

        return model

    def train(self, X_train, y_train, commands, epochs=50, batch_size=32, validation_split=0.15):
        """Train DNN model"""
        self.commands = commands

        # Normalize features
        X_scaled = self.scaler.fit_transform(X_train)

        # Build model
        self.model = self.build_model(X_scaled.shape[1], len(commands))

        print("\nModel Architecture:")
        self.model.summary()

        # Training callbacks
        from tensorflow.keras.callbacks import EarlyStopping, ReduceLROnPlateau

        early_stop = EarlyStopping(
            monitor='val_loss',
            patience=10,
            restore_best_weights=True,
            verbose=1
        )

        reduce_lr = ReduceLROnPlateau(
            monitor='val_loss',
            factor=0.5,
            patience=5,
            min_lr=0.00001,
            verbose=1
        )

        print(f"\nTraining on {len(X_train)} samples...")

        # Train model
        self.history = self.model.fit(
            X_scaled, y_train,
            epochs=epochs,
            batch_size=batch_size,
            validation_split=validation_split,
            callbacks=[early_stop, reduce_lr],
            verbose=1
        )

        print("\nTraining completed")

    def predict(self, X):
        """Predict command for given features"""
        X_scaled = self.scaler.transform(X.reshape(1, -1))

        probabilities = self.model.predict(X_scaled, verbose=0)[0]
        predicted_idx = np.argmax(probabilities)
        confidence = probabilities[predicted_idx]

        predicted_command = self.commands[predicted_idx]

        scores = {self.commands[i]: float(probabilities[i]) for i in range(len(self.commands))}

        return predicted_command, confidence, scores

    def evaluate(self, X_test, y_test):
        """Evaluate model on test set"""
        X_scaled = self.scaler.transform(X_test)

        predictions = self.model.predict(X_scaled, verbose=0)
        predicted_classes = np.argmax(predictions, axis=1)

        accuracy = np.mean(predicted_classes == y_test) * 100

        return accuracy, predicted_classes

    def save_model(self, filepath='recognizer_model.h5'):
        """Save trained model"""
        self.model.save(filepath)

        scaler_data = {
            'scaler': self.scaler,
            'commands': self.commands
        }
        with open('scaler_data.pkl', 'wb') as f:
            pickle.dump(scaler_data, f)

        print(f"Model saved to {filepath}")

    def load_model(self, filepath='recognizer_model.h5'):
        """Load trained model"""
        from tensorflow.keras.models import load_model

        self.model = load_model(filepath)

        with open('scaler_data.pkl', 'rb') as f:
            scaler_data = pickle.load(f)
        self.scaler = scaler_data['scaler']
        self.commands = scaler_data['commands']

        print("Model loaded successfully")

class HomeAutomationController:
    """Control home devices based on recognized speech commands"""

    def __init__(self, recognizer):
        self.recognizer = recognizer
        self.device_states = {
            'lights': 'off',
            'fan': 'off',
            'temperature': 22,
            'door': 'closed',
            'mode': 'normal'
        }

        # Map commands to actions
        self.command_actions = {
            'on': lambda: self._set_device('lights', 'on'),
            'off': lambda: self._set_device('lights', 'off'),
            'up': lambda: self._adjust_temperature(1),
            'down': lambda: self._adjust_temperature(-1),
            'left': lambda: self._set_device('fan', 'off'),
            'right': lambda: self._set_device('fan', 'on'),
            'yes': lambda: self._set_device('door', 'open'),
            'no': lambda: self._set_device('door', 'closed'),
            'stop': lambda: self._emergency_stop(),
            'go': lambda: self._set_mode('active')
        }

    def _set_device(self, device, state):
        self.device_states[device] = state
        return f"{device.capitalize()} turned {state}"

    def _adjust_temperature(self, delta):
        self.device_states['temperature'] += delta
        return f"Temperature set to {self.device_states['temperature']}C"

    def _emergency_stop(self):
        self.device_states['lights'] = 'off'
        self.device_states['fan'] = 'off'
        self.device_states['mode'] = 'standby'
        return "Emergency stop - all devices deactivated"

    def _set_mode(self, mode):
        self.device_states['mode'] = mode
        return f"System mode: {mode}"

    def process_audio(self, audio_path):
        """Process audio file and execute command"""
        feature_extractor = FeatureExtractor()
        mfcc = feature_extractor.extract_mfcc(audio_path)

        if mfcc is None:
            return None

        stat_features = feature_extractor.extract_statistical_features(mfcc)
        predicted_command, confidence, scores = self.recognizer.predict(stat_features)

        # Execute action
        action_result = self.command_actions[predicted_command]()

        return {
            'command': predicted_command,
            'confidence': confidence,
            'action': action_result,
            'state': self.device_states.copy()
        }

    def display_state(self):
        """Display current system state"""
        print("\nCurrent Home Automation State:")
        print("-" * 40)
        for device, state in self.device_states.items():
            if device == 'temperature':
                print(f"  {device.capitalize()}: {state}C")
            else:
                print(f"  {device.capitalize()}: {state.upper()}")
        print("-" * 40)
