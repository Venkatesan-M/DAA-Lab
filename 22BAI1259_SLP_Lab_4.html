# Load movie reviews dataset for sentiment analysis
def load_movie_reviews():
    documents = []
    labels = []

    # Load positive reviews
    for fileid in movie_reviews.fileids('pos'):
        documents.append(movie_reviews.raw(fileid))
        labels.append(1)  # Positive sentiment

    # Load negative reviews
    for fileid in movie_reviews.fileids('neg'):
        documents.append(movie_reviews.raw(fileid))
        labels.append(0)  # Negative sentiment

    return documents, labels

# Text preprocessing function
def preprocess_text(text):
    """
    Comprehensive text preprocessing pipeline
    """
    # Convert to lowercase
    text = text.lower()

    # Remove special characters and digits
    text = re.sub(r'[^a-zA-Z\s]', '', text)

    # Remove extra whitespaces
    text = ' '.join(text.split())

    # Tokenize by splitting into sentences first
    sentences = nltk.sent_tokenize(text, language='english')
    tokens = []
    for sentence in sentences:
        tokens.extend(word_tokenize(sentence, preserve_line=True))


    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    tokens = [token for token in tokens if token not in stop_words]

    # Stemming
    stemmer = PorterStemmer()
    tokens = [stemmer.stem(token) for token in tokens]

    return ' '.join(tokens)

# Apply preprocessing to all texts
print("Preprocessing texts...")
processed_texts = [preprocess_text(text) for text in texts]
print("Preprocessing completed!")

# Implement Bag of Words (BoW) model
class BagOfWordsClassifier:
    def __init__(self, classifier_type='nb', max_features=5000):
        self.max_features = max_features
        self.vectorizer = CountVectorizer(max_features=max_features, binary=True)

        # Initialize classifier based on type
        if classifier_type == 'nb':
            self.classifier = MultinomialNB()
        elif classifier_type == 'lr':
            self.classifier = LogisticRegression(random_state=42)
        else:
            self.classifier = SVC(kernel='linear', random_state=42)

    def fit(self, X_train, y_train):
        # Transform text to BoW features
        X_train_bow = self.vectorizer.fit_transform(X_train)

        # Train classifier
        self.classifier.fit(X_train_bow, y_train)

    def predict(self, X_test):
        # Transform test data and predict
        X_test_bow = self.vectorizer.transform(X_test)
        return self.classifier.predict(X_test_bow)

    def get_feature_names(self):
        return self.vectorizer.get_feature_names_out()

class NgramClassifier:
    def __init__(self, ngram_range=(1, 2), classifier_type='nb', max_features=5000):
        self.ngram_range = ngram_range
        self.max_features = max_features
        self.vectorizer = CountVectorizer(
            ngram_range=ngram_range,
            max_features=max_features
        )

        # Initialize classifier
        if classifier_type == 'nb':
            self.classifier = MultinomialNB()
        elif classifier_type == 'lr':
            self.classifier = LogisticRegression(random_state=42)
        else:
            self.classifier = SVC(kernel='linear', random_state=42)

    def fit(self, X_train, y_train):
        # Transform text to n-gram features
        X_train_ngram = self.vectorizer.fit_transform(X_train)

        # Train classifier
        self.classifier.fit(X_train_ngram, y_train)

    def predict(self, X_test):
        # Transform test data and predict
        X_test_ngram = self.vectorizer.transform(X_test)
        return self.classifier.predict(X_test_ngram)

# Advanced comparison with TF-IDF
class TFIDFClassifier:
    def __init__(self, ngram_range=(1, 1), classifier_type='nb', max_features=5000):
        self.vectorizer = TfidfVectorizer(
            ngram_range=ngram_range,
            max_features=max_features,
            sublinear_tf=True
        )

        if classifier_type == 'nb':
            self.classifier = MultinomialNB()
        elif classifier_type == 'lr':
            self.classifier = LogisticRegression(random_state=42)
        else:
            self.classifier = SVC(kernel='linear', random_state=42)

    def fit(self, X_train, y_train):
        X_train_tfidf = self.vectorizer.fit_transform(X_train)
        self.classifier.fit(X_train_tfidf, y_train)

    def predict(self, X_test):
        X_test_tfidf = self.vectorizer.transform(X_test)
        return self.classifier.predict(X_test_tfidf)

# Comprehensive evaluation and comparison
def evaluate_model(y_true, y_pred, model_name):
    """
    Comprehensive model evaluation
    """
    accuracy = accuracy_score(y_true, y_pred)
    report = classification_report(y_true, y_pred, target_names=['Negative', 'Positive'])
    cm = confusion_matrix(y_true, y_pred)

    print(f"\n{model_name} Results:")
    print(f"Accuracy: {accuracy:.4f}")
    print("\nClassification Report:")
    print(report)

    return accuracy, cm
