# Download and extract Tamil corpus
def download_tamil_corpus():
    """
    Downloads the Tamil corpus from Leipzig Corpora Collection
    This is a well-established linguistic resource for Tamil text analysis
    """
    # Leipzig Corpora Collection - Tamil corpus
    url = "https://downloads.wortschatz-leipzig.de/corpora/tam_mixed_2013_1M.tar.gz"

    try:
        print("Downloading Tamil corpus from Leipzig Corpora Collection...")
        response = requests.get(url, stream=True)

        if response.status_code == 200:
            # Create data directory
            os.makedirs('tamil_corpus', exist_ok=True)

            # Save the file
            with open('tamil_corpus.tar.gz', 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)

            print("Corpus downloaded successfully!")
            return True
        else:
            print(f"Failed to download corpus. Status code: {response.status_code}")
            return False

    except Exception as e:
        print(f"Error downloading corpus: {e}")
        return False

# Tamil word tokenization and analysis
def tokenize_tamil_words(text):
    """
    Tokenizes Tamil text into words
    Tamil uses different script, so we need special handling
    """
    # Tamil unicode range: \u0B80-\u0BFF
    # Split on whitespace and punctuation, keep Tamil characters
    words = re.findall(r'[\u0B80-\u0BFF]+', text)

    # Filter out single characters and very short words
    words = [word for word in words if len(word) > 1]

    return words

def tokenize_tamil_syllables(text):
    """
    Basic syllable tokenization for Tamil
    Tamil syllables typically consist of consonant-vowel combinations
    """
    # Tamil consonants and vowels
    consonants = '[-]'  # Simplified range
    vowels = '[-]'

    # Basic syllable pattern (this is simplified)
    syllable_pattern = r'[\u0B80-\u0BFF][\u0BBE-\u0BC2]?'
    syllables = re.findall(syllable_pattern, text)

    return [syl for syl in syllables if syl.strip()]

# Type-Token analysis for words
def perform_type_token_analysis(tokens, token_type="words"):
    """
    Performs type-token analysis on the given tokens
    """
    # Count token frequencies
    token_counts = Counter(tokens)

    # Calculate statistics
    total_tokens = len(tokens)  # Total number of tokens
    unique_types = len(token_counts)  # Number of unique types
    type_token_ratio = unique_types / total_tokens if total_tokens > 0 else 0

    # Frequency distribution
    freq_dist = list(token_counts.values())
    freq_counter = Counter(freq_dist)

    print(f"\n=== {token_type.upper()} TYPE-TOKEN ANALYSIS ===")
    print(f"Total {token_type}: {total_tokens}")
    print(f"Unique {token_type}: {unique_types}")
    print(f"Type-Token Ratio: {type_token_ratio:.4f}")
    print(f"Most frequent {token_type[:4]}: {token_counts.most_common(10)}")

    return {
        'total_tokens': total_tokens,
        'unique_types': unique_types,
        'type_token_ratio': type_token_ratio,
        'freq_distribution': freq_counter,
        'token_counts': token_counts
    }

# POS tagging comparison analysis
def compare_pos_tagging():
    """
    Compares manual and NLTK POS tagging results
    """
    manual_dict = {word: tag for word, tag in manual_tags}
    nltk_dict = {word: tag for word, tag in nltk_tags if word != '.'}

    matches = 0
    total = len(manual_dict)
    differences = []

    print("\nPOS Tagging Comparison:")
    print(f"{'Word':<15} {'Manual':<8} {'NLTK':<8} {'Match':<8}")
    print("-" * 45)

    for word in manual_dict:
        manual_tag = manual_dict[word]
        nltk_tag = nltk_dict.get(word, 'N/A')
        match = manual_tag == nltk_tag

        if match:
            matches += 1
        else:
            differences.append((word, manual_tag, nltk_tag))

        print(f"{word:<15} {manual_tag:<8} {nltk_tag:<8} {'' if match else '':<8}")

    accuracy = matches / total * 100
    print(f"\nAccuracy: {matches}/{total} = {accuracy:.1f}%")

    if differences:
        print(f"\nDifferences found in {len(differences)} cases:")
        for word, manual, nltk_tag in differences:
            print(f"  {word}: Manual={manual}, NLTK={nltk_tag}")

    return accuracy

def nltk_shallow_parsing():
    """
    NLTK automated shallow parsing using chunk parser
    """
    # Define grammar for chunking
    grammar = r"""
        NP: {<DT|JJ|NN.*>+}          # Noun phrases
        PP: {<IN><NP>}               # Prepositional phrases
        VP: {<VB.*><NP|PP|CLAUSE>*}  # Verb phrases
    """

    # Create chunk parser
    cp = RegexpParser(grammar)

    # Parse the sentence
    sentence = test_sentences[0]
    tokens = word_tokenize(sentence)
    pos_tags = pos_tag(tokens)

    # Remove punctuation for cleaner parsing
    pos_tags = [(word, tag) for word, tag in pos_tags if word != '.']

    # Perform chunking
    chunks = cp.parse(pos_tags)

    print("\nNLTK Shallow Parsing Results:")
    print("="*50)

    def extract_chunks(tree, chunk_list=None):
        if chunk_list is None:
            chunk_list = []

        for subtree in tree:
            if hasattr(subtree, 'label'):
                chunk_words = [word for word, pos in subtree.leaves()]
                chunk_list.append((subtree.label(), chunk_words))
                print(f"{subtree.label()}: {' '.join(chunk_words)}")

        return chunk_list

    nltk_chunks = extract_chunks(chunks)

    # Display tree structure
    print(f"\nParse Tree:")
    chunks.pprint()

    return nltk_chunks, chunks

def process_all_sentences():
    """
    Process all 10 sentences with NLTK POS tagging and shallow parsing
    """
    print("\nProcessing All Test Sentences:")
    print("="*70)

    # Define comprehensive grammar for better chunking
    comprehensive_grammar = r"""
        NP: {<DT|PRP\$>?<JJ>*<NN.*>+}
        PP: {<IN><NP>}
        VP: {<VBZ|VBP|VBD|VBN|VBG|VB|MD><.*>*?<NP|PP>}
        ADVP: {<RB.*>}
        ADJP: {<JJ><JJ>*}
    """

    cp = RegexpParser(comprehensive_grammar)

    results = []

    for i, sentence in enumerate(test_sentences, 1):
        print(f"\nSentence {i}:")
        print(f"Text: {sentence}")

        # Tokenize and POS tag
        tokens = word_tokenize(sentence)
        pos_tags = pos_tag(tokens)

        # Remove punctuation
        pos_tags_clean = [(word, tag) for word, tag in pos_tags if word.isalnum()]

        # Shallow parsing
        parsed = cp.parse(pos_tags_clean)

        # Extract chunks
        chunks = []
        for subtree in parsed:
            if hasattr(subtree, 'label'):
                chunk_words = [word for word, pos in subtree.leaves()]
                chunks.append((subtree.label(), chunk_words))

        results.append({
            'sentence_id': i,
            'word_count': len(tokens),
            'pos_tags': pos_tags,
            'chunks': chunks,
            'parse_tree': parsed
        })

        print(f"Word count: {len(tokens)}")
        print(f"Chunks identified: {len(chunks)}")
        for chunk_type, words in chunks[:3]:  # Show first 3 chunks
            print(f"  {chunk_type}: {' '.join(words)}")

    return results
