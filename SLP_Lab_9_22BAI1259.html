def load_data(file_path):
    """Load and preprocess the translation dataset"""
    with open(file_path, 'r', encoding='utf-8') as file:
        lines = file.read().split('\n')

    pairs = []
    for line in lines[:50000]:  # Limit dataset size for computational efficiency
        if '\t' in line:
            parts = line.split('\t')
            if len(parts) >= 2:
                pairs.append([parts[0].strip(), parts[1].strip()])

    return pairs

def preprocess_text(text):
    """Clean and preprocess text data"""
    text = text.lower()
    text = re.sub(r"[^a-zA-Z0-9\s]", "", text)
    text = re.sub(r"\s+", " ", text)
    return text.strip()

def create_tokenizer(texts):
    """Create word-to-index and index-to-word mappings"""
    word_counts = Counter()
    for text in texts:
        words = text.split()
        word_counts.update(words)

    vocab = ['<PAD>', '<START>', '<END>', '<UNK>'] + [word for word, _ in word_counts.most_common(8000)]
    word_to_index = {word: idx for idx, word in enumerate(vocab)}
    index_to_word = {idx: word for word, idx in word_to_index.items()}

    return word_to_index, index_to_word, len(vocab)

def encode_sequences(texts, word_to_index, max_len):
    """Convert text sequences to numerical sequences"""
    sequences = []
    for text in texts:
        words = text.split()
        sequence = [word_to_index.get(word, word_to_index['<UNK>']) for word in words]
        sequence = sequence[:max_len-1] + [word_to_index['<END>']]
        sequence += [word_to_index['<PAD>']] * (max_len - len(sequence))
        sequences.append(sequence)
    return np.array(sequences)

class Encoder(tf.keras.layers.Layer):
    def __init__(self, vocab_size, embedding_dim, units):
        super(Encoder, self).__init__()
        self.units = units
        self.embedding = Embedding(vocab_size, embedding_dim)
        self.lstm = LSTM(units, return_state=True)

    def call(self, inputs):
        embedded = self.embedding(inputs)
        output, state_h, state_c = self.lstm(embedded)
        return output, state_h, state_c

class Decoder(tf.keras.layers.Layer):
    def __init__(self, vocab_size, embedding_dim, units):
        super(Decoder, self).__init__()
        self.units = units
        self.embedding = Embedding(vocab_size, embedding_dim)
        self.lstm = LSTM(units, return_sequences=True, return_state=True)
        self.dense = Dense(vocab_size, activation='softmax')

    def call(self, inputs, initial_state):
        embedded = self.embedding(inputs)
        output, state_h, state_c = self.lstm(embedded, initial_state=initial_state)
        predictions = self.dense(output)
        return predictions, state_h, state_c

def translate_sequence(input_seq):
    """Translate an input sequence using the trained model"""
    # Encode the input sequence
    states_value = encoder_model.predict(input_seq, verbose=0)

    # Generate target sequence
    target_seq = np.zeros((1, 1))
    target_seq[0, 0] = target_word_to_index['<START>']

    decoded_sentence = []
    stop_condition = False

    while not stop_condition:
        output_tokens, h, c = decoder_model.predict(
            [target_seq] + states_value, verbose=0
        )

        sampled_token_index = np.argmax(output_tokens[0, -1, :])
        sampled_word = target_index_to_word.get(sampled_token_index, '<UNK>')

        if sampled_word == '<END>' or len(decoded_sentence) > max_target_length:
            stop_condition = True
        else:
            decoded_sentence.append(sampled_word)

        target_seq = np.zeros((1, 1))
        target_seq[0, 0] = sampled_token_index
        states_value = [h, c]

    return ' '.join(decoded_sentence)

def translate_text(input_text):
    """Translate input text to target language"""
    # Preprocess input text
    processed_text = preprocess_text(input_text)

    # Convert to sequence
    input_sequence = encode_sequences([processed_text], input_word_to_index, max_input_length)

    # Translate
    translation = translate_sequence(input_sequence)

    return translation

def calculate_bleu_approximation():
    """Simple BLEU score approximation"""
    correct_translations = 0
    total_translations = min(100, len(X_val_enc))

    for i in range(total_translations):
        input_seq = X_val_enc[i:i+1]
        predicted_translation = translate_sequence(input_seq)

        # Get actual target
        actual_target = []
        for idx in X_val_dec[i]:
            if idx == target_word_to_index['<PAD>']:
                break
            word = target_index_to_word.get(idx, '<UNK>')
            if word not in ['<START>', '<END>']:
                actual_target.append(word)

        actual_translation = ' '.join(actual_target)

        # Simple comparison
        if predicted_translation.strip() == actual_translation.strip():
            correct_translations += 1

    bleu_approx = correct_translations / total_translations
    print(f"BLEU Score Approximation: {bleu_approx:.4f}")
    return bleu_approx

def interactive_translation():
    """Interactive translation interface"""
    print("Interactive English-Spanish Translation System")
    print("Type 'quit' to exit")

    while True:
        user_input = input("Enter English text to translate: ")

def batch_translate_file(input_file, output_file):
    """Translate sentences from input file and save to output file"""
    try:
        with open(input_file, 'r', encoding='utf-8') as f:
            sentences = f.readlines()

        translations = []
        for sentence in sentences:
            sentence = sentence.strip()
            if sentence:
                translation = translate_text(sentence)
                translations.append(f"{sentence}\t{translation}")

        with open(output_file, 'w', encoding='utf-8') as f:
            f.write('\n'.join(translations))

        print(f"Batch translation completed. Results saved to {output_file}")

    except FileNotFoundError:
        print(f"Input file {input_file} not found")
    except Exception as e:
        print(f"Error during batch translation: {str(e)}")
